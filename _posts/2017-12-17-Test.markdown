STRING
#include<iostream>
#include<string>
using namespace std;
int main(){
  string s;
  cin>>s;    
  //当有字符出现后遇空白（空格符，换行符，制表符）便终止
  //如输入”sdjajsd  1321weia“  输入的结果将只会为sdjajsd
  cout<<s<<endl;
  return 0;
}
